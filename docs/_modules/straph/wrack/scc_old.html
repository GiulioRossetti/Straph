    bounds = self.get_links_bounds_per_nodes()
    w_components = self.weakly_bounded_connected_components()
    _, l_link_presence = self.get_links_components(w_components, bounds)
    print(" Link according to WBC Components : Done !")
    max_id = 0
    if saving_path:
        pathlib.Path(saving_path).mkdir(parents=True, exist_ok=True)
        max_id = len(next(os.walk(saving_path))[2])
    r = Parallel(n_jobs=8, mmap_mode='r+')(delayed(scc_from_wbc)(link_presence, max_id + i, saving_path)
                                           for i, link_presence in
                                           zip(range(len(l_link_presence)), l_link_presence))
    scc_dag = cdag.merge_scc_dags(r)
    scc_dag.times = self.times
    # Add isolated nodes to the stream
    isolated_nodes = self.get_isolated_nodes()
    print("Nb Isolated nodes :", len(isolated_nodes))
    isolated_nodes = [comp.connected_component(times=[i[1], i[2]],
                                               nodes=set([i[0]]),
                                               ) for i in isolated_nodes]
    scc_dag.add_nodes(isolated_nodes)
    return scc_dag




[docs]def scc_from_wbc(link_presence, id_wbc=None, saving_path=None):
    # code each link, 1 for a beginning, -1 for an ending
    links = []
    scc_stream = cdag.scc_dag()
    for l, lp in link_presence.items():
        for t0, t1 in zip(lp[::2], lp[1::2]):
            links.append(((l[0][0], l[1][0]), t0, t1))  # Little trick overhere (popopo)
            links.append(((l[0][0], l[1][0]), t1))
    # del link_presence   # No more needed
    code = [0] * len(links)
    l = len(links) // 2
    code[::2] = [1] * l
    code[1::2] = [-1] * l
    code = [x for _, x in sorted(zip(links, code), key=lambda y: y[0][1])]
    links.sort(key=lambda y: y[1])
    code = list(reversed(code))
    links = list(reversed(links))
    presents_nodes = set()  # Nodes active at current time.
    local_components = []  # List of current strongly connected components (object)
    dict_node_2_component = {}  # dictionary associating a node to the local current component
    nodes_departure_times = Counter()  # Time of departure for each nodes
    cnt_id = 0
    while links:
        l = links.pop()
        u, v = l[0]
        c = code.pop()
        if c == 1:
            t1 = l[2]
            # logger.debug("Arrival :" + str(l))
            if u not in presents_nodes and v not in presents_nodes:
                # logger.debug("Create Component")
                create_cc(dict_node_2_component, l, local_components, nodes_departure_times,
                          presents_nodes)
            elif u in presents_nodes and v not in presents_nodes:
                # logger.debug("Add a node :" + str(v))
                cnt_id = link_update_cc(scc_stream, dict_node_2_component, l, local_components,
                                        nodes_departure_times, presents_nodes, u, v, saving_path=saving_path,
                                        counter_id=cnt_id, id_wbc=id_wbc)
            elif u not in presents_nodes and v in presents_nodes:
                # logger.debug("Add a node :" + str(u))
                cnt_id = link_update_cc(scc_stream, dict_node_2_component, l, local_components,
                                        nodes_departure_times, presents_nodes, v, u, saving_path=saving_path,
                                        counter_id=cnt_id, id_wbc=id_wbc)
            elif dict_node_2_component[u] != dict_node_2_component[v]:
                # logger.debug("Merge two components")
                cnt_id = merge_cc(scc_stream, dict_node_2_component, l, local_components, nodes_departure_times,
                                  saving_path=saving_path,
                                  counter_id=cnt_id, id_wbc=id_wbc)
            else:
                # logger.debug("Update node's departure times")
                nodes_departure_times[v] = max(t1, nodes_departure_times[v])
                nodes_departure_times[u] = max(t1, nodes_departure_times[u])
                local_components[dict_node_2_component[u]].add_link(l)
                # logger.debug("Node " + str(u) + " departure :" + str(nodes_departure_times[u]))
                # logger.debug("Node " + str(v) + " departure :" + str(nodes_departure_times[v]))
        else:
            # logger.debug("Departure :" + str(l))
            t1 = l[1]
            l = l[0]
            # logger.debug("Node " + str(u) + " departure :" + str(nodes_departure_times[u]))
            # logger.debug("Node " + str(v) + " departure :" + str(nodes_departure_times[v]))
            if nodes_departure_times[u] == t1:
                cnt_id = remove_link_cc(scc_stream, dict_node_2_component, l, local_components,
                                        nodes_departure_times, presents_nodes, t1, u, v, saving_path=saving_path,
                                        counter_id=cnt_id, id_wbc=id_wbc)
            elif nodes_departure_times[v] == t1:
                cnt_id = remove_link_cc(scc_stream, dict_node_2_component, l, local_components,
                                        nodes_departure_times, presents_nodes, t1, v, u, saving_path=saving_path,
                                        counter_id=cnt_id, id_wbc=id_wbc)
            else:
                # logger.debug("Splitting Procedure")
                # In this procedure we can use the BFS to compute any kind of distance, which can be of use further on.
                cnt_id = split_procedure_cc(scc_stream, dict_node_2_component, l, local_components, t1,
                                            saving_path=saving_path,
                                            counter_id=cnt_id, id_wbc=id_wbc)
        # logger.debug("Current components :")
        # for cc in local_components:
        #     logger.debug("\t Times : "+str(cc.times)+" Nodes : "+str(cc.nodes)+" Links : "+str(cc.links))
        # logger.debug("")
    return scc_stream




[docs]def split_procedure_cc(SCC_stream, dict_node_2_component, l, local_components, t1, saving_path=None,
                       counter_id=None, id_wbc=None):
    ncu = dict_node_2_component[l[0]]
    loc_comp = local_components[ncu]
    R = loc_comp.split(l)
    #######################
    # Splitting procedure #
    #######################
    if R:
        if loc_comp.times[0] != t1:
            cop_comp = copy.copy(loc_comp)
            cop_comp.set_end_time(t1)  # Put an end time to the previous component
            if saving_path:
                cop_comp.set_id(str(id_wbc) + "_" + str(counter_id))
                counter_id += 1
                cop_comp.dump_link_presence(saving_path)
            SCC_stream.add_node(cop_comp)  # Add it to the list
        for C in R:
            for n in C.nodes:
                dict_node_2_component[n] = len(local_components)
            C.set_begin_time(t1)  # set new begin time
            local_components.append(C)
        loc_comp.clear()
    return counter_id




[docs]def remove_link_cc(SCC_stream, dict_node_2_component, l, local_components, nodes_departure_times,
                   presents_nodes, t1, u, v, saving_path=None, counter_id=None, id_wbc=None):
    #  By default we suppose that u is leaving.
    # Remove the link, close the component
    if nodes_departure_times[v] == t1:
        # ADD a test for double departure, if a node is already gone, remove the other one.
        # Close component, get rid of these motherfucking nodes
        # logger.debug("\t !Double Departure : " + str(l[0]))
        ncu = dict_node_2_component[v]
        loc_comp = local_components[ncu]
        presents_nodes.discard(v)
        presents_nodes.discard(u)
        if loc_comp.size() <= 2 and loc_comp.nodes:  # If there is less than 2 nodes clear the component
            if loc_comp.times[0] != t1:
                cop_comp = copy.copy(loc_comp)
                cop_comp.set_end_time(t1)  # Put an end time to the previous component
                if saving_path:
                    cop_comp.set_id(str(id_wbc) + "_" + str(counter_id))
                    counter_id += 1
                    cop_comp.dump_link_presence(saving_path)
                SCC_stream.add_node(cop_comp)  # Add it to the list
            loc_comp.clear()
        else:
            # Else double departure means a split
            # logger.debug("\t !!Split!! : " + str(l[0]))
            split_procedure_cc(SCC_stream, dict_node_2_component, l, local_components, t1,
                               saving_path=saving_path, counter_id=counter_id, id_wbc=id_wbc)
    else:
        # logger.debug("Simple departure :" + str(u))
        nckeep = dict_node_2_component[v]
        loc_comp = local_components[nckeep]
        # if loc_comp.times : # ???????????????????? To deal with simultaneous "things"
        if loc_comp.times[0] != t1:
            cop_comp = copy.copy(loc_comp)
            cop_comp.set_end_time(t1)  # Put an end time to the previous component
            if saving_path:
                cop_comp.set_id(str(id_wbc) + "_" + str(counter_id))
                counter_id += 1
                cop_comp.dump_link_presence(saving_path)
            SCC_stream.add_node(cop_comp)  # Add it to the list
        loc_comp.remove_node(u)  # Remove node
        loc_comp.remove_link(l)  # Remove Link
        loc_comp.set_begin_time(t1)  # Set begin time
        presents_nodes.discard(u)
    return counter_id




[docs]def merge_cc(SCC_stream, dict_node_2_component, l, local_components, nodes_departure_times, saving_path=None,
             counter_id=None, id_wbc=None):
    u, v = l[0]
    t0, t1 = l[1], l[2]
    ncu, ncv = dict_node_2_component[u], dict_node_2_component[v]
    loc_comp1 = local_components[ncu]
    if loc_comp1.times[0] != t0:
        cop_comp = copy.copy(loc_comp1)
        cop_comp.set_end_time(t0)  # Put an end time to the previous component
        if saving_path:
            cop_comp.set_id(str(id_wbc) + "_" + str(counter_id))
            counter_id += 1
            cop_comp.dump_link_presence(saving_path)
        SCC_stream.add_node(cop_comp)  # Add it to the list
    loc_comp2 = local_components[ncv]
    if loc_comp2.times[0] != t0:
        cop_comp = copy.copy(loc_comp2)
        cop_comp.set_end_time(t0)  # Put an end time to the previous component
        if saving_path:
            cop_comp.set_id(str(id_wbc) + "_" + str(counter_id))
            counter_id += 1
            cop_comp.dump_link_presence(saving_path)
        SCC_stream.add_node(cop_comp)  # Add it to the list
    loc_comp1.merge(loc_comp2)  # Actualize the 1st component and merge with 2nd one
    loc_comp1.set_begin_time(t0)
    loc_comp1.add_link(l)  # Add the current link
    for node in loc_comp2.nodes:
        dict_node_2_component[node] = ncu  # Actualize referencement before deletion of 2nd comp
    nodes_departure_times[v] = max(t1, nodes_departure_times[v])
    nodes_departure_times[u] = max(t1, nodes_departure_times[u])
    loc_comp2.clear()  # Clear second component
    return counter_id

[docs]def propagation(self, refactor=True, reverse_time=False):
    # Start from a connected component
    #  Step 1: Sort links in their order of appearance.
    # Step 2: Grow each strongly connected component from the previous one, and keep track of minimum departure time.
    # stop if one of the following is true :
    #   C1 : A link appears containing a node from the current component and another
    #   which does not belongs to the component. (entry test) <- Cosntruct new scc with the new node
    #   We only consider actived component, not sure whereas the C2 is mandatory.
    #   C2 : A new link begin at a time superior to the minimum of departure times. (exit time).
    #   <- Construct a component from the new link.
    # Stop the algorithm when the last link is analyzed.

    # Reverse time, pour lancer l'algo sur le stream graph inversé temporellement !
    bounds = self.get_links_bounds_per_nodes()
    Neighborhood = self.get_neighborhood_from_node_partition(bounds)
    w_components = self.weakly_bounded_connected_components(Neighborhood)
    _, l_link_presence = self.get_links_components(w_components, bounds)
    s_components = []  # Composante : Une liste de noeud, un temps de départ (fixe), un temps d'arrêt (provisoire).
    for wcc, link_presence in zip(w_components, l_link_presence):
        # sort links (inverse order, using pop on list)
        if reverse_time:
            link_presence = [((l[0][0], l[1][0]), (t1, t0)) for l, lp in link_presence.items() for (t0, t1) in
                             zip(lp[::2], lp[1::2])]
            link_presence.sort(key=lambda v: v[1][0])
        else:
            link_presence = [((l[0][0], l[1][0]), (t0, t1)) for l, lp in link_presence.items() for (t0, t1) in
                             zip(lp[::2], lp[1::2])]
            link_presence.sort(key=lambda v: v[1][0], reverse=True)

        presents_nodes = set()  # Nodes active at current time.
        local_components = []  # List of current strongly connected components
        dict_node_2_component = {}  # dictionary associating a node to the local current component
        nodes_departure_times = Counter()  # Time of departure for each nodes

        while link_presence:
            l = link_presence.pop()
            u, v = l[0]
            t0, t1 = l[1]
            current_time = t0  # Temps courant, arrivé du noeud.
            # logger.debug("current link :" + str((u, v)) + "time :" + str((t0, t1)))
            # 0. Nouveau temps :
            #   Terminer puis actualiser les composantes ayant un noeud qui se casse, (temps de départ < current_time),
            #    Terminer dans l'ordre croissant des départs
            if nodes_departure_times:
                self.leaving_test(current_time, dict_node_2_component, local_components, nodes_departure_times,
                                  presents_nodes, s_components,
                                  reverse_time=reverse_time)  # while nodes_departure_times and nodes_departure_times.top() < current_time:
            #  Nouveau lien qui arrive !
            # Possibilités :
            # 1. On a vu aucun des deux noeuds  : -> nouvelle composante
            if u not in presents_nodes and v not in presents_nodes:
                self.create_component(dict_node_2_component, local_components, nodes_departure_times,
                                      presents_nodes, t0, t1, u, v)

            # 2. On a vu un des deux noeuds mais pas l'autre :
            # -> terminer puis actualiser la composante correspondante
            elif u in presents_nodes and v not in presents_nodes:
                loc_comp = local_components[dict_node_2_component[u]]
                self.update_component(nodes_departure_times, current_time, dict_node_2_component,
                                      loc_comp, presents_nodes, s_components,
                                      u, v, t1, reverse_time=reverse_time)

            elif u not in presents_nodes and v in presents_nodes:
                loc_comp = local_components[dict_node_2_component[v]]
                self.update_component(nodes_departure_times, current_time, dict_node_2_component,
                                      loc_comp, presents_nodes, s_components,
                                      v, u, t1, reverse_time=reverse_time)

            # 3. On a vu les deux noeuds, 2 possibilités :
            #   3.1 Les deux noeuds sont dans la même composantes <- actualiser les temps de départ
            elif dict_node_2_component[u] == dict_node_2_component[v]:
                if reverse_time:
                    nodes_departure_times[v] = min(t1, nodes_departure_times[v])
                    nodes_departure_times[u] = min(t1, nodes_departure_times[u])
                else:
                    nodes_departure_times[v] = max(t1, nodes_departure_times[v])
                    nodes_departure_times[u] = max(t1, nodes_departure_times[u])

            #   3.2 Les deux noeuds sont dans des composantes différentes :
            #  <- terminer les deux composantes puis les fusionner
            else:
                self.merge_and_update_components(current_time, dict_node_2_component, local_components,
                                                 nodes_departure_times, s_components, t1, u, v,
                                                 reverse_time=reverse_time)

        # 4. deal with the reamining disappearation times
        if reverse_time:
            for d_time, n in sorted(zip(nodes_departure_times.values(), nodes_departure_times.keys()),
                                    reverse=True):
                loc_comp = local_components[dict_node_2_component[n]]
                if loc_comp[1] != d_time:
                    loc_comp[2] = d_time
                    s_components.append([copy.copy(i) for i in loc_comp])
                loc_comp[0].discard(n)
                loc_comp[1] = d_time
        else:
            for d_time, n in sorted(zip(nodes_departure_times.values(), nodes_departure_times.keys())):
                loc_comp = local_components[dict_node_2_component[n]]
                if loc_comp[1] != d_time:
                    loc_comp[2] = d_time
                    s_components.append([copy.copy(i) for i in loc_comp])
                loc_comp[0].discard(n)
                loc_comp[1] = d_time
    # REFORMAT
    if refactor and reverse_time:
        s_components = [[(n, scc[2], scc[1]) for n in scc[0]] for scc in s_components]
    if refactor and not reverse_time:
        s_components = [[(n, scc[1], scc[2]) for n in scc[0]] for scc in s_components]
    return s_components



[docs]def leaving_test(self, current_time, dict_node_2_component, local_components, nodes_departure_times, presents_nodes,
                 s_components, reverse_time):
    if reverse_time:
        disp_times = [(n, t) for n, t in nodes_departure_times.items() if t > current_time]
        disp_times.sort(key=lambda x: x[1], reverse=True)
    else:
        disp_times = [(n, t) for n, t in nodes_departure_times.items() if t < current_time]
        disp_times.sort(key=lambda x: x[1])
    for n, d_time in disp_times:
        loc_comp = local_components[dict_node_2_component[n]]
        if loc_comp[1] != d_time:
            loc_comp[2] = d_time
            s_components.append([copy.copy(i) for i in loc_comp])
        loc_comp[0].discard(n)
        loc_comp[1] = d_time
        presents_nodes.discard(n)
        del nodes_departure_times[n]



[docs]def merge_and_update_components(self, current_time, dict_node_2_component, local_components, nodes_departure_times,
                                s_components, t1, u, v, reverse_time):
    loc_comp1 = local_components[dict_node_2_component[u]]
    loc_comp1[2] = current_time  # Put an end time to 1st scc
    s_components.append([copy.copy(i) for i in loc_comp1])  # Add it to the list

    s_components.append([copy.copy(i) for i in loc_comp2])  # Add it to the list
    loc_comp1[0] |= loc_comp2[0]
    # Actualize the 1st component and merge with 2nd one
    loc_comp1[1] = current_time  # Input a new begining time
    for node in loc_comp2[0]:
        dict_node_2_component[node] = dict_node_2_component[u]  # Actualize referencement
    # del local_components[dict_node_2_component[v]]      # Del 2nd component
    if reverse_time:
        nodes_departure_times[v] = min(t1, nodes_departure_times[v])
        nodes_departure_times[u] = min(t1, nodes_departure_times[u])
    else:
        nodes_departure_times[v] = max(t1, nodes_departure_times[v])
        nodes_departure_times[u] = max(t1, nodes_departure_times[u])

    s_components = []  # Composante : Une liste de noeud, un temps de départ (fixe), un temps d'arrêt (provisoire)
    # , une liste de liens
    for wcc, link_presence in zip(w_components, l_link_presence):
        # code each link, 1 for a beginning, -1 for an ending
        links = []
        for l, lp in link_presence.items():
            for t0, t1 in zip(lp[::2], lp[1::2]):
                links.append(((l[0][0], l[1][0]), t0, t1))  # Little trick overhere
                links.append(((l[0][0], l[1][0]), t1))
        code = [0] * len(links)
        l = len(links) // 2
        code[::2] = [1] * l
        code[1::2] = [-1] * l
        code = [x for _, x in sorted(zip(links, code), key=lambda y: y[0][1])]
        links.sort(key=lambda y: y[1])
        code = list(reversed(code))
        links = list(reversed(links))

        presents_nodes = set()  # Nodes active at current time.
        local_components = []  # List of current strongly connected components
        dict_node_2_component = {}  # dictionary associating a node to the local current component
        nodes_departure_times = Counter()  # Time of departure for each nodes
        l_components_links = []
        while links:
            l = links.pop()
            u, v = l[0]
            c = code.pop()
            if c == 1:
                t1 = l[2]
                # logger.debug("Arrival :"+str(l))
                #  En cas d'arrivée, 4 sous cas :
                # 1. Creation
                # 2. Add a node
                # 3. Merge
                # 4. Update times
                if u not in presents_nodes and v not in presents_nodes:
                    # logger.debug("Create Component")
                    self.create_component_v2(l, dict_node_2_component, local_components,
                                             nodes_departure_times, presents_nodes, l_components_links)

                elif u in presents_nodes and v not in presents_nodes:
                    # logger.debug("Add a node :"+str(v))
                    self.add_node_scc(l, v, u, dict_node_2_component, local_components, nodes_departure_times,
                                      presents_nodes, s_components, l_components_links)

                elif u not in presents_nodes and v in presents_nodes:
                    # logger.debug("Add a node :"+str(u))
                    self.add_node_scc(l, u, v, dict_node_2_component, local_components, nodes_departure_times,
                                      presents_nodes, s_components, l_components_links)

                elif dict_node_2_component[u] != dict_node_2_component[v]:
                    # logger.debug("Merge two components")
                    self.merge_and_update_components_v2(l, dict_node_2_component, local_components,
                                                        nodes_departure_times, s_components, l_components_links)
                else:
                    # logger.debug("Update node's departure times")
                    nodes_departure_times[v] = max(t1, nodes_departure_times[v])
                    nodes_departure_times[u] = max(t1, nodes_departure_times[u])
                    # logger.debug("Node " + str(u) + " departure :" + str(nodes_departure_times[u]))
                    # logger.debug("Node " + str(v) + " departure :" + str(nodes_departure_times[v]))
                    l_components_links[dict_node_2_component[u]].add(l[0])
            else:
                # logger.debug("Departure :"+str(l))
                # En cas de départ, 2 sous cas:
                # 1. simple départ d'un des noeuds (check set des départs)
                # 2. Split de la composante, lancer une BFS sur une des extrémités,
                # 2.1 Si l'autre extrémité est dans la résultante component, do nothing.
                # 2.2 sinon conserver le résultat et lancer une autre BFS sur l'autre extrémité <- 2 composantes.
                t1 = l[1]
                # logger.debug("Node "+str(u)+" departure :" + str(nodes_departure_times[u]))
                # logger.debug("Node "+str(v)+" departure :" + str(nodes_departure_times[v]))
                #######################
                # BAIL : BLACK MAGIC  #
                #######################
                if nodes_departure_times[u] == t1:
                    self.remove_node_scc(l, u, v, dict_node_2_component, local_components, nodes_departure_times,
                                         presents_nodes,
                                         s_components, l_components_links)
                elif nodes_departure_times[v] == t1:
                    self.remove_node_scc(l, v, u, dict_node_2_component, local_components, nodes_departure_times,
                                         presents_nodes,
                                         s_components, l_components_links)
                else:
                    # logger.debug("Splitting Procedure")
                    self.splitting_procedure(l, dict_node_2_component, local_components, s_components,
                                             l_components_links)

        #     logger.debug("current SCC :" + str(s_components))
        #     logger.debug("\n")
        # logger.debug("#####################################")
    if refactor:
        s_components = [[(n, scc[1], scc[2]) for n in scc[0]] for scc in s_components]
    if scc_with_links:
        # PIMP DAT SHIT
        return s_components, l_components_links
    else:
        return s_components

[docs]def BFS_scc(self, edgelist, u, v, special_split=False):
    # CUSTOM BFS
    # In a component which is being splitted, the output can
    # only be 1 or 2 components.
    # If at any moment u and v find themselves in the same component <- stop
    comp_bfs = []
    link_comp_bfs = []
    present_nodes_bfs = set()
    dict_node_2_component_bfs = {}
    for l in edgelist:
        n1, n2 = l
        # logger.debug("\t current link : "+str(l))
        if n1 not in present_nodes_bfs and n2 not in present_nodes_bfs:
            # logger.debug("\t Create comp")
            dict_node_2_component_bfs[n1] = len(comp_bfs)
            dict_node_2_component_bfs[n2] = len(comp_bfs)
            comp_bfs.append(set([n1, n2]))
            link_comp_bfs.append(set([l]))
            present_nodes_bfs.add(n1)
            present_nodes_bfs.add(n2)

        elif n1 in present_nodes_bfs and n2 not in present_nodes_bfs:
            # logger.debug("\t ADD "+str(n2)+" in "+str(n1)+" component")
            nc1 = dict_node_2_component_bfs[n1]
            comp_bfs[nc1].add(n2)
            link_comp_bfs[nc1].add(l)
            dict_node_2_component_bfs[n2] = nc1
            present_nodes_bfs.add(n2)

        elif n1 not in present_nodes_bfs and n2 in present_nodes_bfs:
            # logger.debug("\t ADD "+str(n1)+" in "+str(n2)+" component")
            nc2 = dict_node_2_component_bfs[n2]
            comp_bfs[nc2].add(n1)
            link_comp_bfs[nc2].add(l)
            dict_node_2_component_bfs[n1] = nc2
            present_nodes_bfs.add(n1)

        elif dict_node_2_component_bfs[n1] != dict_node_2_component_bfs[n2]:
            # logger.debug("\t Merge "+str(n2)+" component into "+str(n1)+" component")
            nc1, nc2 = dict_node_2_component_bfs[n1], dict_node_2_component_bfs[n2]
            comp_bfs[nc1] |= comp_bfs[nc2]
            link_comp_bfs[nc1] |= link_comp_bfs[nc2]
            link_comp_bfs[nc1].add(l)
            link_comp_bfs[nc2].clear()
            for node in comp_bfs[nc2]:
                dict_node_2_component_bfs[node] = nc1
            comp_bfs[nc2].clear()
        else:
            link_comp_bfs[dict_node_2_component_bfs[n1]].add(l)
        # Test if we can stop the bfs_with_path
        if u in dict_node_2_component_bfs and v in dict_node_2_component_bfs and \
                dict_node_2_component_bfs[u] == dict_node_2_component_bfs[v]:
            # logger.debug("\t Test succesfull : " + str(u) + "is with " + str(v))
            return False
    # logger.debug("\t comp bfs_with_path : " + str(comp_bfs))
    # logger.debug("\t link bfs_with_path : " + str(link_comp_bfs))
    if not special_split:
        ncu, ncv = dict_node_2_component_bfs[u], dict_node_2_component_bfs[v]
        return comp_bfs[ncu], link_comp_bfs[ncu], comp_bfs[ncv], link_comp_bfs[ncv]
    else:
        return comp_bfs, link_comp_bfs
