[docs]def get_source_nodes(S, high_degree_nodes, random_nodes):
    if random_nodes == True:
        source_nodes = random.sample(S.nodes, 10)
        source_nodes = [(t0, t1, n) for n, npres in zip(S.nodes, S.node_presence)
                        for t0, t1 in zip(npres[::2], npres[1::2]) if n in source_nodes]
    elif high_degree_nodes:
        d = S.degrees_partition()
        source_nodes = set()
        for degree, n in sorted(d.items(), reverse=True):
            for i in n:
                source_nodes.add(i[2])
                if len(source_nodes) == 10:
                    break
            if len(source_nodes) == 10:
                break
        source_nodes = [(t0, t1, n) for n, npres in zip(S.nodes, S.node_presence)
                        for t0, t1 in zip(npres[::2], npres[1::2]) if n in source_nodes]
    else:
        source_nodes = [(t0, t1, n) for n, npres in zip(S.nodes, S.node_presence)
                        for t0, t1 in zip(npres[::2], npres[1::2])]
    return source_nodes

                print("\nconfig :",config)
                print("entry :",entry_format)
                print("delta :",delta)
                csv_file = list(p.glob('*.csv'))[0]
                file_name = csv_file.stem
                out_path = p / file_name
                print("csv file :",csv_file)
                print("out path :",out_path)
                sort_csv(csv_file, entry_format=entry_format, **config)
                parser(input_file=csv_file, input_format='csv',
                       links_duration = delta,
                       entry_format=entry_format, output_file=str(out_path),
                       output_format=['sg'], **config)





[docs]def dataset_description(dataset_directory):
    # Properties to collect must be specified and computed manually
    dict_data = {'n':[],
                 'm':[],
                 'N':[],
                 'M':[],
                 '|T_S|': [],
                 'd_{max}':[],
                 'd_{mean}':[]
                 }
    dataset_paths = {}
    dir = pathlib.Path(dataset_directory)
    for p in dir.iterdir():
        if p.is_dir():
            csv_file = list(p.glob('*.csv'))[0]
            dataset_name = csv_file.stem
            dataset_path = str(p)+"/"
            dataset_paths[dataset_name] = dataset_path
    done_datasets = []
    for ds in dataset_paths:
        S = load_dataset(dataset_paths[ds],ds)
        new_node_presence = [[S.times[0], S.times[1]] for _ in S.nodes]
        S.node_presence = new_node_presence
        print("Dataset Loaded :",ds)
        S.describe()
        if sum([len(item) / 2 for item in S.link_presence]) >= 100000:
            print("Skip Dataset")
            continue
        ### general description ###
        n = len(S.nodes)
        N = sum([len(item) / 2 for item in S.node_presence])
        m = len(S.links)
        M = sum([len(item) / 2 for item in S.link_presence])

        ## SCC ###
        dag = S.condensation_dag()
        n_SCC = len(dag.c_nodes)

        ## Interactions Times (snpashot) ##
        T_S = set()
        it = S.get_interaction_times()
        for u in it:
            T_S |= set(it[u])
        T_S = len(T_S)

        ## Degrees ##
        degrees_partition = S.degrees_partition(interaction_times=it)
        d_max = max(degrees_partition.keys())
        d_mean,tot = 0,0
        for d,v in degrees_partition.items():
            for u in v:
                d_mean += (u[1]-u[0])*d
                tot += u[1]-u[0]
        d_mean = d_mean/tot

        dict_data['n'].append(n)
        dict_data['N'].append(N)
        dict_data['m'].append(m)
        dict_data['M'].append(M)
        dict_data['|T_S|'].append(T_S)
        dict_data['d_{mean}'].append(d_mean)
        dict_data['d_{max}'].append(d_max)

        done_datasets.append(ds)
    Index = pd.Index(done_datasets)
    D = pd.DataFrame(dict_data, index=Index)
    with open("/home/leo/Dev/Data_Stream/description_dataset.csv", 'w') as hdl:
        D.to_csv(hdl,na_rep="None")
    return D
