[docs]def L_Algorithm(S, source, L_functions):
    '''
    An implementation of L-Algorithm (described in ~add ref)
    :param S: A Stream Graph
    :param source: A temporal source node
    :param L_functions: Functions according to the path's type (supported :
            - foremost path
            - shortest foremost path
            - shortest path
            - fastest shortest path
            - fastest path
            - shortest fastest path
    :return:
    '''
    N = S.segmented_neighborhood()
    J = set()  # set of visited segmented links
    queue = SortedSet()
    # Initialisation :
    L = L_functions['initialisation'](source)
    queue.add((source[0], source))
    while queue:
        tu, u = queue.pop(0)
        visited_neihgbors = set()
        for e in N[u]:
            t0, t1, v = e
            if (t0, t1, u, v) not in J:
                # We will not visit links that connect u and v several time, because they are dominated...
                # Do not apply for fastest paths, as we can leave later to obtain a fastest path.
                if tu <= t1:  # If we can take the link
                    J.add((t0, t1, u, v))
                    J.add((t0, t1, v, u))
                    # Get the best distances and arrival times compatible with the link !
                    # A better distance for 'v' can exist but with a later arrival time which isn't compatible with the current link.
                    L_functions['update'](L, t0, t1, v, u)
                    L_functions['update'](L, t0, t1, u, v)
                    # We must remove useless elements from L[v]:
                    # Every path that is a longer than a previous one and that arrives after or at the same time
                    # But we can keep longer paths that arrives before (in case of a shortcut link not yet analyzed)
                    L_functions['domination'](L, v)
                    L_functions['domination'](L, u)
                    if v not in visited_neihgbors:
                        av = max(tu, t0)
                        queue.add((av, v))
                        visited_neihgbors.add(v)
    return L

[docs]def FoP_initialisation(source):
    '''
    'L' initialisation for FoP
    :param source:
    :return:
    '''
    L = defaultdict(SortedSet)
    L[source].add(source[0])  # (a_u)
    return L




[docs]def FoP_update(L, t0, t1, u, v):
    '''
    We update 'L' according to the link (t0,t1,u,v) and FoP properties.
    :param L:
    :param t0:
    :param t1:
    :param u:
    :param v:
    :return:
    '''
    last_element_before_link = None
    for au in L[u]:
        if au <= t0:  # We have just added the last element of L[u] before the beginning of the link
            last_element_before_link = t0
        elif t0 < au <= t1:  #  We take arrival times compatible with the link.
            L[v].add(au)
        elif au > t1:  # L is ordered by a_u, we can discard further elements
            break
    if last_element_before_link is not None:
        L[v].add(last_element_before_link)




[docs]def FoP_domination(L, v):
    '''
    We remove FoP-dominated elements from L.
    :param L:
    :param v:
    :return:
    '''
    to_remove = set()
    pred_av = L[v][0]
    for i in range(1, len(L[v])):
        av = L[v][i]
        if av >= pred_av:
            to_remove.add(av)
        else:
            pred_av = av
    L[v] -= to_remove




[docs]def FoP_postprocess(L, source):
    '''
    Post process elements in 'L' to obtain time to reach 'ttr'
    :param L:
    :param source:
    :return:
    '''
    ttr = {}
    for k, v in L.items():
        n = k[2]
        if n in ttr:
            ttr[n] = min(ttr[n], v[0] - source[0])
        else:
            ttr[n] = v[0] - source[0]
    return ttr




[docs]def FoP_ss(S, source):
    '''
    Return the time to reach every reachable node in *S* from *source*.
    :param S:
    :param source:
    :return:
    '''
    L_functions = {'initialisation': FoP_initialisation,
                   'update': FoP_update,
                   'domination': FoP_domination}
    L = L_Algorithm(S, source, L_functions)
    times_to_reach = FoP_postprocess(L, source)
    return times_to_reach

[docs]def SFoP_update(L, t0, t1, u, v):
    last_element_before_link = None
    for au, du in L[u]:
        if au <= t0:  # We have just added the last element of L[u] before the beginning of the link
            last_element_before_link = (t0, du + 1)
        elif t0 < au <= t1:  #  We take arrival times compatible with the link.
            L[v].add((au, du + 1))
        elif au > t1:  # L is ordered by a_u, we can discard further elements
            break
    if last_element_before_link is not None:
        L[v].add(last_element_before_link)




[docs]def SFoP_domination(L, v):
    to_remove = set()
    pred_av, pred_dv = L[v][0]
    for i in range(1, len(L[v])):
        Lv = L[v][i]
        av, dv = Lv
        if (av > pred_av and dv >= pred_dv) or (av == pred_av and dv > pred_dv):
            to_remove.add(Lv)
        else:
            pred_av = av
            pred_dv = dv
    L[v] -= to_remove




[docs]def SFoP_postprocess(L, source):
    ttr = {}
    lengths = {}
    for k, v in L.items():
        n = k[2]
        duration = v[0][0]
        if n in lengths:
            lengths[n] = min(min([i[1] for i in v if i[0] == duration]), lengths[n])
            ttr[n] = min(ttr[n], duration - source[0])
        else:
            lengths[n] = min([i[1] for i in v if i[0] == duration])
            ttr[n] = duration - source[0]
    return ttr, lengths

[docs]def SP_update(L, t0, t1, u, v):
    # The Function is efficient, improvement ? : bissect in order to only add elements between t1 and t0
    # and the last element with a_i <= t0
    for (du, au) in L[u]:
        if au <= t1:  #  We take arrival times compatible with the link.
            new_arrival = au
            new_path = du + 1
            if au <= t0:  # We have just added the last element of L[u] before the beginning of the link given the ordering of L.
                L[v].add((new_path, t0))
                break
            L[v].add((new_path, new_arrival))

[docs]def FSP_postprocess(L):
    distances = {}
    durations = {}
    for k, v in L.items():
        n = k[2]
        length = v[0][0]
        min_duration = min([i[1] - i[2] for i in v if i[0] == length])
        if min_duration < 0:
            min_duration = 0

        if n in distances:
            distances[n] = min(distances[n], length)
            durations[n] = min(durations[n], min_duration)
        else:
            distances[n] = length
            durations[n] = min_duration
    return distances, durations

[docs]def SFP_postprocess(L):
    latencies = {}
    length = {}
    for k, v in L.items():
        n = k[2]
        duration = min([i[0] - i[1] for i in v])
        min_length = min([i[2] for i in v if i[0] - i[1] == duration])
        if duration < 0:
            duration = 0

        if n in latencies:
            latencies[n] = min(latencies[n], duration)
            length[n] = min(length[n], min_length)
        else:
            latencies[n] = duration
            length[n] = min_length
    return latencies, length
