[docs]def F_Algorithm_extended(S, L_functions):
    '''
    An implementation of F-Algorithm (add ref)
    Pairwise Algorithm to compute temporal paths in Stream Graph
    :param S: A Stream Graph
    :param L_functions: Functions according to the path's type (supported :
        - foremost path
        - shortest foremost path
        - shortest path
        - fastest shortest path
        - fastest path
        - shortest fastest path)
    :return:
    '''
    # Initialisation
    F = {}
    for n,np in zip(S.nodes,S.node_presence):
        for t0,t1 in zip(np[::2],np[1::2]):
            s = (t0,t1,n)
            F[s] = L_functions['initialisation_extended'](s)
    temporal_adjacency_list = defaultdict(set) # currently present links in the form of adjacency list
    nodes_to_udpate = set() # Reached nodes so far
    E = S.augmented_ordered_links()
    for i in E:
        c = i[0]
        if c == 1: # Arrival
            _, t0, t1, u, v = i
            l = (t0, t1, u, v)
            temporal_adjacency_list[u].add((t1, v))
            temporal_adjacency_list[v].add((t1, u))
            nodes_to_udpate.add(u)
            nodes_to_udpate.add(v)
            F_update_extended(F, [l], temporal_adjacency_list, nodes_to_udpate, L_functions)
        else:
            _, t1, u, v = i
            temporal_adjacency_list[u].remove((t1, v))
            temporal_adjacency_list[v].remove((t1, u))
    return F

[docs]def L_Algorithm_extended(S, source, L_functions):
    '''
    An implementation of L-Algorithm (described in ~add ref)
    Single Source Algorithm to compute temporal paths in Stream Graph
    :param S: A Stream Graph
    :param source: A temporal source node
    :param L_functions: Functions according to the path's type (supported :
            - foremost path
            - shortest foremost path
            - shortest path
            - fastest shortest path
            - fastest path
            - shortest fastest path)
    :return:
    # IMPORTANT : We suppose that we are in the WCC of 'source' otherwise it's fucking expensive !
    '''
    L = L_functions['initialisation_extended'](source)
    temporal_adjacency_list = defaultdict(set)  # key : temporal node, value : (temporal node, end time of link)
    E = S.augmented_ordered_links()
    for i in E:
        c = i[0]
        if c == 1:  # ARRIVAL
            _, t0, t1, u, v = i
            if t1 >= source[0]:
                l = (t0, t1, u, v)
                temporal_adjacency_list[u].add((t1, v))
                temporal_adjacency_list[v].add((t1, u))
                bfs_update_extended(L, source, [l], temporal_adjacency_list, L_functions)
        else:  # DEPARTURE
            _, t1, u, v = i
            if t1 >= source[0]:
                temporal_adjacency_list[u].remove((t1, v))
                temporal_adjacency_list[v].remove((t1, u))
    return L





[docs]def bfs_update_extended(L,source, batch_arrival, temporal_adjacency_list, L_functions):
    '''
    Proceeds to browse every links present at instant :math:'t_0' in order to propagate the update on current possible paths

    :param L:
    :param u:
    :param v:
    :param t0:
    :param t1:
    :param temporal_adjacency_list:
    :param L_functions:
    :return:
    '''
    b = batch_arrival[0][0]
    Q = deque()
    visited = set()
    for (_, e, u1, u2) in batch_arrival:
        if u2 in L and u1 != source:
            Q.append((u2, u1, e))
            visited.add((u2,u1,e))
        if u1 in L and u2 != source:
            Q.append((u1, u2, e))
            visited.add((u1,u2,e))
    if Q:
        while Q:
            pred, w, end_t = Q.popleft()
            updated = L_functions['update_extended'](L, b, end_t, pred, w)
            if updated:
                # removed= L_functions['domination'](L, w)
                for t, k in temporal_adjacency_list[w]:
                    if (w,k,t) not in visited and k != source:
                        visited.add((w,k,t))
                        Q.append((w, k, t))

[docs]def FoP_update_extended(L, t0, t1, u, v):
    '''
    We update 'L' according to the link (t0,t1,u,v) and FoP properties.

    :param L:
    :param t0:
    :param t1:
    :param u:
    :param v:
    :return:
    '''
    new_arrival = t0
    if v in L:
        av = L[v][-1][0]
        if av == new_arrival:
            # Get sum of paths in u :
            cv = 0
            for au,pu,cu in L[u]:
                cv +=cu
            L[v].add((new_arrival,u,cv))
        else:
            return False
    else:
        # Get sum of paths in u :
        cv = 0
        for au,pu,cu in L[u]:
            cv +=cu
        L[v].add((new_arrival,u,cv))
    return True



[docs]def FoP_postprocess_extended(L, source):
    '''
    Post process elements in 'L' to obtain time to reach 'ttr'

    :param L:
    :param source:
    :return:
    '''
    ttr = {}
    cnt = {}
    pred = {}
    for k, v in L.items():
        n = k[2]
        for av,pv,cv in v:
            if n in ttr:
                t = av - source[0]
                if t < ttr[n]:
                    ttr[n] = t
                    cnt[n] = cv
                    pred[n] = set([pv])
                elif t == ttr[n]:
                    cnt[n] +=cv
                    pred[n].add(pv)
            else:
                ttr[n] = av - source[0]
                cnt[n] = cv
                pred[n] = set([pv])
    return ttr,cnt,pred

[docs]def SFoP_update_extended(L, t0, t1, u, v):
    au, du = L[u][-1]
    new_arrival = t0
    new_distance = du + 1
    if v in L:
        av, dv = L[v][-1]
        if (new_arrival, new_distance) not in L[v] and (new_arrival == av and new_distance < dv):  #  Domination Step Here
            done = False
            while not done:
                if new_arrival == av:
                    L[v].remove((av, dv))
                    if L[v]:
                        av, dv = L[v][-1]
                    else:
                        done = True
                else:
                    done = True
            L[v].add((new_arrival, new_distance))
        else:
            # We did not update L : don't propagate the path !
            return False
    else:
        L[v].add((new_arrival, new_distance))
    return True




[docs]def SFoP_postprocess_extended(L, source):
    ttr = {}
    lengths = {}
    for k, v in L.items():
        n = k[2]
        duration = v[0][0]
        if n in lengths:
            lengths[n] = min(min([i[1] for i in v if i[0] == duration]), lengths[n])
            ttr[n] = min(ttr[n], duration - source[0])
        else:
            lengths[n] = min([i[1] for i in v if i[0] == duration])
            ttr[n] = duration - source[0]
    return ttr, lengths

[docs]def FP_update_extended(L, t0, t1, u, v):
    '''
    Lu is sorted by the value of au and we analyze temporal links in temporal order.
    :param L:
    :param t0:
    :param t1:
    :param u:
    :param v:
    :return:
    '''
    au, su = L[u][-1]
    new_arrival = t0
    new_start = min(t1, su)  # Remember that the aim is to leaving the source node the latest possible (thus s_u
    #  can be > to t_1).
    if v in L:
        av, sv = L[v][-1]
        if (new_arrival, new_start) not in L[v] and new_start > sv:  #  Domination Step Here
            done = False
            while not done:
                if av == new_arrival:# or max(av-sv,0) > max(new_arrival-new_start,0):
                    L[v].remove((av, sv))
                    if L[v]:
                        av, sv = L[v][-1]
                    else:
                        done = True
                else:
                    done = True
            L[v].add((new_arrival, new_start))
        else:
            # We did not update L : don't propagate the path !
            return False
    else:
        L[v].add((new_arrival, new_start))
    return True

[docs]def SFP_update_extended(L, t0, t1, u, v):
    '''
    Update Lv from (t0,t1,u,v) according to SFP constraints.

    :param L:
    :param t0:
    :param t1:
    :param u:
    :param v:
    :return:
    '''
    au, su, du = L[u][-1]
    new_arrival = t0
    new_start = min(t1, su)
    new_distance = du + 1
    if v in L:
        av, sv, dv = L[v][-1]
        if (new_arrival, new_start, new_distance) not in L[v] and\
                (new_start > sv or (new_start == sv and new_distance < dv)):
            done = False
            while not done:
                if av == new_arrival:
                    L[v].remove((av, sv, dv))
                    if L[v]:
                        av, sv, dv = L[v][-1]
                    else:
                        done = True
                else:
                    done = True
            L[v].add((new_arrival, new_start, new_distance))
        else:
            return False
    else:
        L[v].add((new_arrival, new_start, new_distance))
    return True





[docs]def SFP_postprocess_extended(L):
    latencies = {}
    length = {}
    for k, v in L.items():
        n = k[2]

        duration = min([i[0] - i[1] for i in v])
        if duration < 0:
            duration = 0
        min_length = min([i[2] for i in v if i[0] - i[1] == duration or i[1] > i[0]]) # Or condition for null latencies
        if n in latencies:
            latencies[n] = min(latencies[n], duration)
            length[n] = min(length[n], min_length)
        else:
            latencies[n] = duration
            length[n] = min_length
    return latencies, length
