[docs]def extract_counter_nodes_links(file_input):
    counter_number_packets_per_node = Counter()
    counter_number_packets_per_links = Counter()
    compteur = 0
    with open(file_input, 'r') as input:
        reader = csv.reader(input, delimiter=';')
        for i in reader:
            compteur += 1
            # Ignore header
            if compteur == 1:
                continue
            line = i
            if line[1] and line[2]:
                counter_number_packets_per_node[line[1]] += 1
                counter_number_packets_per_node[line[2]] += 1
                if (line[2], line[1]) in counter_number_packets_per_links:
                    counter_number_packets_per_links[(line[2], line[1])] += 1
                else:
                    counter_number_packets_per_links[(line[1], line[2])] += 1
    return counter_number_packets_per_node, counter_number_packets_per_links

[docs]def expected_node_degrees(d, nodes, node_presence, expected_degrees=None):
    if not expected_degrees:
        expected_degrees = Counter()
    for n, np in zip(nodes, node_presence):
        expected_degrees[n] = (d[n]) / sum_presence(np)
    return expected_degrees

[docs]def plot_period(anomalies, ax, y=None):
    t_min = min([min(int(i['begin']), int(i['end'])) for i in anomalies if int(i['end']) != 0 and int(i['begin']) != 0])
    # print(t_min)
    for i in anomalies:
        b = int(i['begin']) - t_min
        e = int(i['end']) - t_min
        if b != 0 and (e - b) < 100:  # and len(i['info']) >= 2:
            # print("period :",(e-b), " begin :",b," end :",e)
            if y:
                ax.bar((b + (e - b) / 2), y, width=(e - b), color='#7a1f5c', alpha=0.2)
            else:
                ax.bar((b + (e - b) / 2), width=(e - b), color='#7a1f5c', alpha=0.2)
    return ax
