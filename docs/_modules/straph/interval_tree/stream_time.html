[docs]def get_adjacency_matrix(l_nodes, links, dict_nodes_index=None):
    A = np.zeros((l_nodes, l_nodes))
    if dict_nodes_index:
        for u, v in links:
            A[dict_nodes_index[u], dict_nodes_index[v]] = 1
            A[dict_nodes_index[v], dict_nodes_index[u]] = 1
    else:
        for u, v in links:
            A[u, v] = 1
            A[v, u] = 1
    return A




[docs]def find_neighbors(Neighborhood, start, neighbors=set()):
    neighbors.add(start)
    for n in Neighborhood[start]:
        if n in neighbors:
            continue
        else:
            neighbors.add(n)
            neighbors.update(find_neighbors(Neighborhood, n, neighbors))
    return neighbors

        if not E_t and t:
            E_t = self.get_E_t(t)
        if not V_t and t:
            V_t = self.get_V_t(t)
        possible_pairs = V_t * (V_t - 1) / 2
        if possible_pairs == 0:
            return 0
        return E_t / possible_pairs

[docs]    def instantaneous_clustering_coef(self, t):
        cc = Counter()
        N = self.get_instant_neighborhood(t)
        for v in N:
            v_neighbors = list(N[v])
            l = len(v_neighbors)
            possible_pairs_neighbors = l * (l - 1) / 2
            if possible_pairs_neighbors == 0:
                cc[v] = 0
            else:
                triplet = 0
                for n1 in range(l):
                    for n2 in range(n1 + 1, l):
                        if v_neighbors[n2] in N[v_neighbors[n1]]:
                            triplet += 1
                cc[v] = triplet / possible_pairs_neighbors
        return cc

[docs]    def Katz_measure(self, t, alpha=0.5):
        nodes = self.node_presence.search(t)['elements']
        links = self.link_presence.search(t)['elements']
        l_nodes = len(nodes)
        dict_nodes_index = {}
        for v, i in zip(nodes, range(l_nodes)):
            dict_nodes_index[v] = i
        A = get_adjacency_matrix(l_nodes, links, dict_nodes_index)
        C_Katz = Counter()
        A_p = A
        alpha_p = alpha
        for k in range(10):
            for v, i in zip(nodes, range(l_nodes)):
                C_Katz[v] += (alpha_p) * A_p[i, :].sum()
            A_p = A_p.dot(A)
            alpha_p *= alpha
        return C_Katz



[docs]    def PPR(self, t, alpha=0.2):
        nodes = self.node_presence.search(t)['elements']
        N = self.get_instant_neighborhood(t)
        prr = Counter()
        current_node = random.choice(nodes)
        for k in range(50 * len(nodes)):
            u = random.random()
            # Probability 1-alpha, we move to one of the neighbors
            if u > alpha:
                try:                # To deal with isolated nodes
                    current_node = random.choice(list(N[current_node]))
                    prr[current_node] += 1
                except:
                    pass
            # Probability alpha, random restart
            else:
                current_node = random.choice(nodes)
                prr[current_node] += 1
        return prr

        nodes = self.node_presence.search(t)['elements']
        l_nodes = len(nodes)
        print("Nb of nodes :", l_nodes)
        D = np.full((l_nodes, l_nodes), np.inf)
        dict_nodes_index = {}
        if betweeness:
            B = {v: 0 for v in range(l_nodes)}
        for v, n in zip(range(l_nodes), nodes):
            D[v, v] = 0
            dict_nodes_index[n] = v
        links = self.link_presence.search(t)['elements']
        for u, v in links:
            D[dict_nodes_index[u], dict_nodes_index[v]] = 1
            D[dict_nodes_index[v], dict_nodes_index[u]] = 1
        for k in range(l_nodes):
            for i in range(l_nodes):
                for j in range(l_nodes):
                    if D[i, j] > D[i, k] + D[k, j]:
                        D[i, j] = D[i, k] + D[k, j]
                        if betweeness:
                            B[k] += 1 / D[i, j]
        R = pd.DataFrame(D, columns=nodes, index=nodes)
        if betweeness:
            B = {v: B[i] for v, i in zip(nodes, range(l_nodes))}
            return R, B
        else:
            return R



[docs]    def find_connected_components(self, t):
        nodes = self.node_presence.search(t)['elements']
        N = self.get_instant_neighborhood(t)
        components = []
        i = 0
        isolated_nodes = [i for i in N if not N[i]]
        visited = set(isolated_nodes)
        unvisited = set(nodes) - visited
        # print("Init visited :",len(visited))
        # print("Init Unvisited :",len(unvisited))
        while len(visited) < len(nodes):
            start = random.choice(list(unvisited))
            current_component = find_neighbors(N, start)
            # print("C",i,": ",current_component)
            visited.update(current_component)
            # print("Len Visited :",len(visited))
            components.append(copy.deepcopy(current_component))
            unvisited = unvisited - visited
            # print("Len Unvisited :",len(unvisited))
            current_component.clear()
            i+=1
        if isolated_nodes:
            for i in isolated_nodes:
                components.append(i)
        return components




[docs]    def find_k_core(self,t):
        degrees = self.get_instant_degrees(t)
        N = self.get_instant_neighborhood(t)
        m_d = max(degrees.values())
        D = [[v for v in degrees if degrees[v] == i] for i in range(m_d+1)]
        k = 0
        L =[]
        L_tmp = []
        for j in range(len(degrees)):
            for d in range(m_d+1):
                if D[d]:
                    i = d
                    break
            if i > k:
                k = i
                L.append(L_tmp)
                L_tmp = []
            v = random.choice(D[i])
            L_tmp += [v]
            D[i].remove(v)
            for w in N[v]:
                if w not in itertools.chain.from_iterable(L) and w not in L_tmp:
                    D[degrees[w]].remove(w)
                    degrees[w] -= 1
                    D[degrees[w]].append(w)
        L.append(L_tmp)
        return L,k
