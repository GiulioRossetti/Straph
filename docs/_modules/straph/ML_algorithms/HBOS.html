[docs]def compute_score(i,hist_list,max_hist_list):
    #Normalize hist :
    s = sum([log(1/(len(c_bins)/(max(c_bins)-min(c_bins))/m))
             if len(set(c_bins)) !=1
             else log(1/(len(c_bins)/m))
             for h,m,v in zip(hist_list,max_hist_list,i) for c_bins in h if v in c_bins
             ])
    return s

[docs]def HBOS(df):
    scores=[]
    hist_list=[]
    N= len(df.values)
    k=round(N**(0.5))
    ratio= N/k
    for d in df.columns:
        print("Dimension :",d)
        hist_list.append(compute_hist(df[d].values,ratio))
    max_hist_list=[max([len(c_bins)/(max(c_bins)-min(c_bins))
                        if len(set(c_bins))!=1
                        else len(c_bins)
                        for c_bins in h
                        ])
                   for h in hist_list
                   ]
    # print(max_hist_list)
    for i in df.values:
        # print("Instance :",i)
        scores.append(compute_score(i,hist_list,max_hist_list))
    return scores
