[docs]def random_node_presence(T, mean_flow_nodes, mean_node_presence):
    # Tirage Aléatoire pour la redondance du noeud X ~ (Poisson(moyenne des redondances))
    rep = max(np.random.poisson(mean_flow_nodes), 1)

    #  Pour chaque intervalle de présence, on tire une durée aléatoire
    # X ~ (Uniform(0,2*moyenne des durées des intervalle)) E[X] = moyenne des durées des intervalles
    # dur = list(np.random.uniform(0, 2 * mean_node_presence, rep))
    dur = list(np.random.poisson(mean_node_presence, rep))
    # Placement aléatoire des intervalles sur la fenêtre de temps, fusion des intervalles si nécessaire
    # X ~ (Uniform(T0,T1))
    plac = list(np.sort(np.random.uniform(T[0], T[1], rep)))
    acc = plac[0] + dur[0]
    if acc > T[1]:
        # Si on dépasse la borne supérieure de temps de notre Stream Graph
        acc = T[1]
        n_presence = [plac[0], acc]
        return n_presence
    n_presence = [plac[0], acc]  #  Initialisation, [t0,t0+d0]
    # print("n_presence :", n_presence)
    for i in range(1, rep):
        acc = plac[i] + dur[i]
        if acc > T[1]:
            # Si on dépasse la borne supérieure de temps de notre Stream Graph
            acc = T[1]
        if acc <= n_presence[-1]:
            # Cas ou l'intervalle est inclus dans le précédent
            continue
        if plac[i] <= n_presence[-1]:
            # Cas de l'intersection : on fusionne
            del n_presence[-1]
            n_presence += [acc]
        else:
            # Cas disjoint
            n_presence += [plac[i], acc]
            # print("n_presence :", n_presence)
    return n_presence




[docs]def random_link_presence(intersec, mean_flow_links, mean_link_presence):
    # Tirage Aléatoire pour la redondance du lien X ~ (Poisson(moyenne des redondances))
    rep = max(np.random.poisson(mean_flow_links), 1)
    #  Pour chaque intervalle de présence, on tire une durée aléatoire
    # X ~ (Uniform(0,2*moyenne des durées des intervalle)) E[X] = moyenne des durées des intervalles
    # dur = list(np.random.uniform(0, 2 * mean_link_presence, rep))
    dur = list(np.random.poisson(mean_link_presence, rep))
    # Placement aléatoire des intervalles sur les fenêtres d'intersection, fusion des intervalles si nécessaire
    # X ~ (Uniform(T0,T1))
    plac = []
    id_intersec = {}  #  Pour retenir l'intervalle d'intersection ou se situe le lien (Black MAGIC)
    for r in range(rep):
        i = 2 * random.choice(range(len(intersec) // 2))  # On choisit aléatoirement un temps commun aux 2 noeuds
        T0 = intersec[i]
        T1 = intersec[i + 1]
        plac.append(np.random.uniform(T0, T1))
        id_intersec[plac[-1]] = i + 1
    plac.sort()
    # Initialisation avant check puis fusion si necessaire
    acc = plac[0] + dur[0]
    if acc > intersec[id_intersec[plac[0]]]:
        acc = intersec[id_intersec[plac[0]]]
    l_presence = [plac[0], acc]

    for i in range(1, rep):
        acc = plac[i] + dur[i]
        if acc > intersec[id_intersec[plac[i]]]:
            # Si on dépasse la borne supérieure de temps de notre fenêtre
            acc = intersec[id_intersec[plac[i]]]
        if acc <= l_presence[-1]:
            # Cas ou l'intervalle est inclus dans le précédent (cas d'un accroissement faible)
            continue
        if plac[i] <= l_presence[-1]:
            # Cas de l'intersection : on fusionne
            del l_presence[-1]
            l_presence += [acc]
        else:
            # Cas disjoint
            l_presence += [plac[i], acc]
    return l_presence




[docs]def get_intersection(u, v, node_presence):
    intersec = []
    for ut0, ut1 in zip(node_presence[u][::2], node_presence[u][1::2]):
        for vt0, vt1 in zip(node_presence[v][::2], node_presence[v][1::2]):
            if ut0 <= vt1 and vt0 <= ut1:
                intersec += [max(ut0, vt0), min(vt1, ut1)]
    return intersec




[docs]def random_link(u, v, p_link):
    unif = random.random()
    if unif < p_link:
        return (u, v)
    return False

[docs]def generate_stream_graph(T,
                          nb_nodes,
                          mean_flow_nodes,
                          mean_node_presence,
                          p_link,
                          mean_flow_links,
                          mean_link_presence
                          ):
    # Initialisation des noeuds
    nodes = list(range(nb_nodes))
    # Initialisation de la présence des noeuds
    node_presence = [[] for n in nodes]
    links = []
    link_presence = []
    for n in nodes:
        node_presence[n] = random_node_presence(T, mean_flow_nodes, mean_node_presence)
    for i in range(nb_nodes):
        u = nodes[i]
        for j in range(i + 1, nb_nodes):
            v = nodes[j]
            current_link = random_link(u, v, p_link)
            if current_link:
                intersec = get_intersection(u, v, node_presence)
                if intersec:
                    links.append(current_link)
                    link_presence.append(random_link_presence(intersec,
                                                              mean_flow_links,
                                                              mean_link_presence))
    print("Nodes : ", len(nodes))
    print("Node presence : ", sum((len(n) for n in node_presence)))
    print("Links : ", len(links))
    print("Link presence : ", sum((len(l) for l in link_presence)))
    S = sg.stream_graph(times=T,
                        nodes=nodes,
                        node_presence=node_presence,
                        links=links,
                        link_presence=link_presence,
                        )
    S.set_card_W()
    S.set_card_E()
    print("|W| :", S.card_W)
    print("|E| :", S.card_E)
    return S




[docs]def generate_kcore_example(nb_nodes, k):
    N = [i for i in range(nb_nodes)]
    N_presence = [[0, nb_nodes] for i in range(1, nb_nodes + 1)]
    L = [(i, i + 1) for i in range(nb_nodes - 1)]
    L_presence = [[0, i] for i in range(1, len(L) + 1)]
    d = 1 / nb_nodes
    if k > 1:
        for c in range(2, k + 1):
            L += [(i, i + c)
                  if i + c < nb_nodes
                  else (i, (i + c) % (nb_nodes))
                  for i in range(nb_nodes - 1)]
            for i in range(nb_nodes - 1):
                L_presence.append([d / nb_nodes, d])
                d += 1 / nb_nodes

    T = [0, nb_nodes]
    S = sg.stream_graph(times=T,
                        nodes=N,
                        node_presence=N_presence,
                        links=L,
                        link_presence=L_presence)
    print("Nodes :", len(N), " n_prez :", len(N_presence))
    print("Links :", len(L), " l_prez :", len(L_presence))
    return S
