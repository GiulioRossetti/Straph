[docs]def compute_wcc_from_links(links_input, wcc_file=None):
    comp_2_nb_links = {}  # Dictionary associating a comp to its number of current temporal links
    node_2_status = {}  # Dictionary associating a node to his current status : (current degree, id current comp)
    local_components = []  # List of local components
    cnt_wcc_id = 0  # Id of stored wcc
    cnt_test = 0  # In purpose of printing
    t_begin = time.time()
    ###########
    gc.disable()
    ###########
    with open(links_input, 'rb') as input:
        unpacker = msgpack.Unpacker(input, use_list=False)
        for i in unpacker:
            cnt_test += 1
            if cnt_test % 1000000 == 0:
                print(" i :", i, " t :", time.time() - t_begin)
            c = i[0]
            l = i[1:]
            if c == 1:
                u, v = l[2],l[3]
                if u not in node_2_status and v not in node_2_status:
                    # print("CREATE")
                    create_wcc(l, node_2_status, local_components, comp_2_nb_links)
                    # assert node_2_status[u][1] == node_2_status[v][1]

                elif u not in node_2_status and v in node_2_status:
                    # print("UPDATE from :", v)
                    update_wcc(v, u, l, node_2_status, local_components, comp_2_nb_links)
                    # assert node_2_status[u][1] == node_2_status[v][1]

                elif u in node_2_status and v not in node_2_status:
                    # print("UPDATE from :", u)
                    update_wcc(u, v, l, node_2_status, local_components, comp_2_nb_links)
                    # assert node_2_status[u][1] == node_2_status[v][1]

                elif node_2_status[u][1] != node_2_status[v][1]:
                    # print("MERGE")
                    merge_wccs(l, node_2_status, local_components, comp_2_nb_links)
                    # assert node_2_status[u][1] == node_2_status[v][1]

                else:
                    node_2_status[u][0] += 1
                    node_2_status[v][0] += 1
                    n_current_comp = node_2_status[u][1]
                    # print("Update TIME in comp :", n_current_comp)
                    comp_2_nb_links[n_current_comp] += 1
                    comp = local_components[n_current_comp]
                    comp.links.append(l)
                    comp.active_links.add((u,v))
                    assert node_2_status[u][1] == node_2_status[v][1]
            else:
                u, v = l[1],l[2]
                assert node_2_status[u][1] == node_2_status[v][1]
                n_current_comp = node_2_status[u][1]
                node_2_status[u][0] -= 1
                node_2_status[v][0] -= 1
                comp_2_nb_links[n_current_comp] -= 1

                comp = local_components[n_current_comp]
                comp.active_links.remove((u,v))
                if comp_2_nb_links[n_current_comp] == 0:
                    # print("STORE COMP :", n_current_comp)
                    cnt_wcc_id = end_comp(n_current_comp, local_components, cnt_wcc_id, wcc_file)

                if node_2_status[u][0] == 0:
                    # print("Remove node :", u)
                    del node_2_status[u]
                if node_2_status[v][0] == 0:
                    # print("Remove node :", v)
                    del node_2_status[v]
    # Store the rest of the wcc
    for n_comp in range(len(local_components)):
        if local_components[n_comp]:
            # print("STORE COMP")
            cnt_wcc_id = end_comp(n_comp, local_components, cnt_wcc_id, wcc_file)
    ###########
    gc.enable()
    ###########
    return




[docs]def end_comp(n_current_comp, local_components, cnt_wcc_id, storage_path=None):
    '''
    Store the component
    :param n_current_comp:
    :param local_components:
    :param storage_path:
    :return:
    '''
    comp_to_store = local_components[n_current_comp]
    comp_to_store.id = cnt_wcc_id
    wcc.store_wcc_to_sgf(comp_to_store, storage_path)
    local_components[n_current_comp] = None
    cnt_wcc_id += 1
    return cnt_wcc_id




[docs]def create_wcc(l, node_2_status, local_components, comp_2_nb_links):
    '''
    Create a wcc component
    :param u:
    :param v:
    :param t0:
    :param t1:
    :param node_2_status:
    :param nb_comp:
    :param comp_2_nb_links:
    :return:
    '''
    u, v = l[2],l[3]
    n_comp = len(local_components)
    node_2_status[u] = [1, n_comp]
    node_2_status[v] = [1, n_comp]
    comp_2_nb_links[n_comp] = 1
    local_components.append(wcc.weakly_connected_component(links=[l],active_links=set([(u,v)])))




[docs]def update_wcc(node_to_update, node_to_add, l, node_2_status, local_components, comp_2_nb_links):
    '''
    Update a component with a new node
    :param node_to_update:
    :param node_to_add:
    :param t0:
    :param t1:
    :param node_2_status:
    :param local_components:
    :param comp_2_nb_links:
    :return:
    '''
    u,v = l[2],l[3]
    n_current_comp = node_2_status[node_to_update][1]
    comp_2_nb_links[n_current_comp] += 1
    node_2_status[node_to_update][0] += 1
    node_2_status[node_to_add] = [1, n_current_comp]

    current_comp = local_components[n_current_comp]
    current_comp.links.append(l)
    current_comp.active_links.add((u,v))




[docs]def merge_wccs(l, node_2_status, local_components, comp_2_nb_links):
    '''
    Merge two components
    :param l:
    :param node_2_status:
    :param local_components:
    :param comp_2_nb_links:
    :return:
    '''
    u, v = l[2],l[3]
    n_comp_1, n_comp_2 = node_2_status[u][1], node_2_status[v][1]
    comp_1 = local_components[n_comp_1]
    comp_2 = local_components[n_comp_2]

    comp_1.links += comp_2.links
    comp_1.active_links |= comp_2.active_links

    comp_1.links.append(l)
    comp_1.active_links.add((u,v))

    for h in comp_2.active_links:
        n, m = h
        if n in node_2_status and m in node_2_status:
            node_2_status[m][1] = n_comp_1
            node_2_status[n][1] = n_comp_1

    node_2_status[u][0] += 1
    node_2_status[v][0] += 1
    comp_2_nb_links[n_comp_1] += comp_2_nb_links[n_comp_2] + 1

    del comp_2_nb_links[n_comp_2]
    local_components[n_comp_2] = None

    node_2_status[v][1] = n_comp_1




[docs]def check_intersection(dict_node_presence,u,t0,t1,id_wcc):
    '''
    Check u is already in dict_node_presence between t0 and t1 in another wcc
    :param dict_node_presence:
    :param u:
    :param t0:
    :param t1:
    :param id_wcc:
    :return:
    '''
    for j in dict_node_presence[u]:
        if j != id_wcc:
            for b,e in zip(dict_node_presence[u][j][::2],dict_node_presence[u][j][1::2]):
                if b<t1 and t0<e:
                    return True
    return False

    '''

    :param wcc_storage_path:
    :return:

    dict_node_presence = {}
    with open(wcc_storage_path,'rb') as ipt:
        unpacker = msgpack.Unpacker(ipt,use_list=False)
        for i in unpacker:
            id_wcc = i[0]
            for j in i[3]:
                if j[0] ==1:
                    t0,t1 =j[1],j[2]
                    u,v =  j[3],j[4]
                    if u in dict_node_presence:
                        if check_intersection(dict_node_presence,u,t0,t1,id_wcc):
                            print("id wcc :",id_wcc)
                            print(" node :",u)
                            print(" at time :",t0,t1)
                            print(" already present at : ",dict_node_presence[u])
                            raise ValueError("Overlap Between WCC")
                        if id_wcc in dict_node_presence[u]:
                            dict_node_presence[u][id_wcc]+=[t0,t1]
                        else:
                            dict_node_presence[u][id_wcc] = [t0,t1]
                    else:
                        dict_node_presence[u] = {id_wcc:[t0,t1]}
                    if v in dict_node_presence:
                        if check_intersection(dict_node_presence,v,t0,t1,id_wcc):
                            print("id wcc :",id_wcc)
                            print(" node :", v)
                            print(" at time :", t0, t1)
                            print(" already present at : ", dict_node_presence[v])
                            raise ValueError("Overlap Between WCC")
                        if id_wcc in dict_node_presence[v]:

                        else:
                            dict_node_presence[v][id_wcc] = [t0,t1]
